# game/state_machine.py class GameEngine: async def night_vote(self, room_id, voter_id, target_id): key = f"room:{room_id}:night_votes" await self.redis.hset(key, voter_id, target_id) mafia = await self.get_mafia_players(room_id) votes = await self.redis.hgetall(key) if len(votes) == len(mafia): victim = Counter(votes.values()).most_common(1)[0][0] await self.process_night_result(room_id, victim) await self.broadcast(room_id, { "event": "night_end", "victim": victim, })
